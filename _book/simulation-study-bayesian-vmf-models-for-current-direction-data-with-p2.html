6.2 Bayesian Inference on a conjugate prior with p=2.

Used as an benchmark. See Forbes & Mardia (2015) by \(\texttt{circglmbayes}\).

For the von Mises distribution on \([0,2\pi)\), \(f(\theta\mid\mu,\kappa) = \frac{1}{2\pi I_0(\kappa)}\exp\{\kappa\cos(\theta-\mu)\}\), and conjugate prior \(\pi(\mu,\kappa) \propto I_0(\kappa)^{-c}\exp\{\kappa R_0\cos(\mu-\mu_0)\},\) with hyperparameters \(c\ge0\), \(R_0\ge0\), and \(\mu_0\). The resulting joint posterior is\[\pi(\mu,\kappa\mid\theta_{1:n}) \propto I_0(\kappa)^{-(c+n)}\exp{\kappa R_n\cos(\mu-\bar\theta_n)}\] where the posterior resultant length \(R_n\) and mean direction \(\bar\theta_n\) are defined by \(R_n\cos\bar\theta_n = R_0\cos\mu_0 + \sum_{i=1}^{n}\cos\theta_i\) and \(R_n\sin\bar\theta_n = R_0\sin\mu_0 + \sum_{i=1}^{n}\sin\theta_i\).

We sample from the posterior using a two‑step Gibbs sampler.
Given current values \((\mu,\kappa)\),

  • Draw \(\mu\mid\kappa,\theta\) from a von Mises distribution with mean direction \(\bar\theta_n\) and concentration \(\kappa R_n\).

  • Draw \(\kappa\mid\mu,\theta\) from the full conditional \[ \pi(\kappa\mid\mu,\theta) \propto I_0(\kappa)^{-\eta}\exp\{-\eta\beta_0\kappa\}, \] where \(\eta = c+n\) and \(\beta_0 = -\frac{R_n}{\eta}\cos(\mu-\bar\theta_n)\).

Here \(\mu \in [0,2\pi]\) as an angle. \(\kappa \geq0\) as concentration. Also, \(\texttt{theta}\) as an vector with each scalar \(\theta_i\in [0, 2\pi]\) and dimension n.

plot function

plot_trace_sideways_density <- function(
    paths,
    samples,
    iter = NULL,
    ref_value = NULL,
    ref_density_fun = NULL,
    xlab = "Forward step i",
    ylab = expression(hat(mu)[n+i]),
    main = NULL,
    trace_col = rgb(0.25, 0.45, 0.75, 0.06),
    trace_lwd = 0.5,
    density_fill = rgb(0.25, 0.45, 0.75, 0.30),
    density_border = rgb(0.20, 0.35, 0.65),
    density_lwd = 1.2,
    ref_col = "red",
    ref_lwd = 1.5,
    ref_lty = 2,
    ref_density_col = "red",
    ref_density_lwd = 2,
    density_width_frac = 0.22,
    density_x_offset_frac = 1.02,
    xlim_multiplier = 1.25,
    y_pad_frac = 0.04,
    cex_lab = 1.2,
    cex_axis = 1.0,
    cex_main = 1.2,
    cex_leg = 0.9,
    legend_pos = "topleft",
    show_legend = TRUE,
    show_traj_legend = FALSE,
    mark_initial = TRUE,
    initial_col = "red",
    initial_pch = 19,
    initial_cex = 1.2,
    initial_label = "initial point",
    n_value = 500,
    n_pos = "bottomright",
    n_cex = 0.8,
    n_col = "gray40",
    n_pad_x = 0.02,   
    n_pad_y = 0.03    
) {
  if (!is.matrix(paths)) paths <- as.matrix(paths)
  if (!is.numeric(paths)) stop("`paths` must be numeric.")
  if (!is.numeric(samples)) stop("`samples` must be numeric.")

  if (is.null(iter)) iter <- seq_len(ncol(paths)) - 1
  if (length(iter) != ncol(paths)) {
    stop("`iter` must have length equal to `ncol(paths)`.")
  }

  x_start <- min(iter)
  x_end <- max(iter)
  x_width <- x_end - x_start
  if (x_width <= 0) stop("`iter` must span a positive range.")

  y_range <- range(c(as.vector(paths), samples, ref_value), finite = TRUE)
  y_pad <- y_pad_frac * diff(y_range)
  if (!is.finite(y_pad) || y_pad == 0) y_pad <- 0.1
  y_range <- y_range + c(-y_pad, y_pad)

  dens <- density(samples, n = 512)
  y_grid <- seq(y_range[1], y_range[2], length.out = 500)

  ref_density <- NULL
  if (!is.null(ref_density_fun)) ref_density <- ref_density_fun(y_grid)

  dens_scale <- x_width * density_width_frac / max(dens$y, ref_density, na.rm = TRUE)
  x_offset <- x_start + x_width * density_x_offset_frac
  x_max <- x_start + x_width * xlim_multiplier

  plot(
    NULL,
    xlim = c(x_start, x_max),
    ylim = y_range,
    xlab = "",
    ylab = ylab,
    main = main,
    xaxt = "n",
    cex.lab = cex_lab,
    cex.axis = cex_axis,
    cex.main = cex_main
  )

  ticks <- pretty(c(x_start, x_end))
  ticks <- ticks[ticks >= x_start & ticks <= x_end]
  axis(1, at = ticks, labels = ticks, lwd = 0, lwd.ticks = 1, cex.axis = cex_axis)
  usr <- par("usr")
  segments(x_start, usr[3], x_end, usr[3], xpd = FALSE)
  mtext(xlab, side = 1, line = 2.8, at = (x_start + x_end) / 2, cex = cex_lab)

  for (i in seq_len(nrow(paths))) {
    lines(iter, paths[i, ], col = trace_col, lwd = trace_lwd)
  }

  if (!is.null(ref_value)) {
    abline(h = ref_value, col = ref_col, lwd = ref_lwd, lty = ref_lty)
  }

  polygon(
    x_offset + dens$y * dens_scale,
    dens$x,
    col = density_fill,
    border = density_border,
    lwd = density_lwd
  )

  if (!is.null(ref_density)) {
    lines(
      x_offset + ref_density * dens_scale,
      y_grid,
      col = ref_density_col,
      lwd = ref_density_lwd
    )
  }

  segments(x_offset, y_range[1], x_offset, y_range[2],
           col = "gray60", lty = 3, lwd = 0.8)

  if (mark_initial) {
    init_x <- iter[1]
    init_y <- paths[1, 1]
    points(init_x, init_y, col = initial_col, pch = initial_pch, cex = initial_cex)
  }

  
  if (!is.null(n_value)) {
    usr <- par("usr")   # c(xmin, xmax, ymin, ymax)
    x_range <- usr[2] - usr[1]
    y_range_cur <- usr[4] - usr[3]

    if (n_pos == "bottomright") {
      x_pos <- usr[2] - x_range * n_pad_x   
      y_pos <- usr[3] + y_range_cur * n_pad_y   
      adj <- c(1, 0)
    } else if (n_pos == "bottomleft") {
      x_pos <- usr[1] + x_range * n_pad_x
      y_pos <- usr[3] + y_range_cur * n_pad_y
      adj <- c(0, 0)
    } else {
      warning("n_pos should be 'bottomright' or 'bottomleft'. Defaulting to bottomright.")
      x_pos <- usr[2] - x_range * n_pad_x
      y_pos <- usr[3] + y_range_cur * n_pad_y
      adj <- c(1, 0)
    }

    text(x = x_pos, y = y_pos, labels = paste0("n = ", n_value),
         adj = adj, cex = n_cex, col = n_col)
  }

  if (show_legend) {
    leg_text <- c("True value", "MPS density", "MCMC density")
    leg_col  <- c(ref_col, density_border, ref_density_col)
    leg_lty  <- c(ref_lty, 1, 1)
    leg_lwd  <- c(ref_lwd, density_lwd, ref_density_lwd)
    leg_pch  <- c(NA, 15, NA)
    leg_pt.cex <- c(NA, 1.5, NA)

    if (show_traj_legend) {
      leg_text <- c("MPS trajectories", leg_text)
      leg_col  <- c(adjustcolor(trace_col, alpha.f = 0.6), leg_col)
      leg_lty  <- c(1, leg_lty)
      leg_lwd  <- c(1.5, leg_lwd)
      leg_pch  <- c(NA, leg_pch)
      leg_pt.cex <- c(NA, leg_pt.cex)
    }

    if (mark_initial) {
      leg_text <- c(leg_text, initial_label)
      leg_col  <- c(leg_col, initial_col)
      leg_lty  <- c(leg_lty, NA)
      leg_lwd  <- c(leg_lwd, NA)
      leg_pch  <- c(leg_pch, initial_pch)
      leg_pt.cex <- c(leg_pt.cex, 1.2)
    }

    legend(
      legend_pos,
      legend = leg_text,
      col    = leg_col,
      lty    = leg_lty,
      lwd    = leg_lwd,
      pch    = leg_pch,
      pt.cex = leg_pt.cex,
      bty    = "n",
      cex    = cex_leg
    )
  }

  invisible(list(
    density = dens,
    y_range = y_range,
    x_offset = x_offset,
    dens_scale = dens_scale
  ))
}

Example for drawing graphes.


set.seed(2026) #for n=100,500

#Settings
nn_cores <- max(detectCores()-1,1)
mu_true <- 2
kappa_true <- 4
n <- 500 # use n=500 for another graph
data <- generate_vmf_data(n, mu_true = mu_true, kappa_true = kappa_true)

#MCMC
res_MCMC <- fit_mcmc_vmf(data$theta,n_iter=7000,burn = 5000)
kappa_MCMC <- res_MCMC[,2]
kappa_den <- density(kappa_MCMC)
ref_fun <- approxfun(kappa_den$x, kappa_den$y, yleft = 0, yright = 0)

#MPS
cor_res <- run_cor_predresamp_vmf(data$X, B = 2000, M = 1000, n_cores = nn_cores, return_chains = TRUE)
kappa_paths <- t(sapply(cor_res$chain, function(chain) {
  sqrt(colSums(chain$c_chain^2))
}))
c0 <- bayesdir::mle_vmf(data$X)   
kappa0 <- sqrt(sum(c0^2))
kappa_paths_with_init <- cbind(kappa0, kappa_paths)
kappa_last <- kappa_paths_with_init[, ncol(kappa_paths_with_init)]

#Draw graph, Finally!
plot_trace_sideways_density(
  paths          = kappa_paths_with_init,
  samples        = kappa_last,          
  iter           = 1:1001,
  ref_value      = kappa_true,          
  ref_density_fun = ref_fun,            
  ref_density_col = "blue",             
  ref_density_lwd = 2,                  
  xlab = "Forward step",
  ylab = "kappa",
  cex_leg = 0.7,
  n_value = n,
  n_cex = 0.9
)
