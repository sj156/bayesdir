5 Predictive Resampler

5.1 The main sampler

The main Gibbs sampler is called run_predresamp_vmf().

#' Simulate Raw Score Path (Step 1 of Algorithm 1)
#' @param c0 Initial estimator (MLE)
#' @param n Initial sample size
#' @param M Depth (iterations)
#' @param save_raw Whether to save the trajectory of c_n,m
#' @return A list containing the raw difference D, quadratic variation Q, and optional path
#' @export
simulate_raw_path <- function(c0, n, M, save_raw = FALSE) {
  p <- length(c0)
  c_curr <- c0
  Q_mat <- matrix(0, p, p)
  
  if (save_raw) {
    c_chain <- matrix(0, p, M + 1)
    c_chain[, 1] <- c0
  }
  
  for (m in 1:M) {
    gamma_m <- 1 / (n + m)
    
    # 1. Draw X_new ~ vMF(c_curr)
    x_new <- rstiefel::rmf.vector(c_curr)
    
    # 2. Compute score s
    s <- score_fn(c_curr, x_new, p)
    
    # 3. Accumulate quadratic variation Q
    Q_mat <- Q_mat + (gamma_m^2) * (s %*% t(s))
    
    # 4. Raw score update
    c_curr <- c_curr + gamma_m * s
    
    if (save_raw) {
      c_chain[, m + 1] <- c_curr
    }
  }
  
  res <- list(
    D_raw = c_curr - c0, # Raw difference (c_n,M - c_n,0)
    Q_nM  = Q_mat,
    c_final_raw = c_curr
  )
  
  if (save_raw) res$c_chain <- c_chain
  return(res)
}

#' Score-Only Martingale Posterior Sampler for vMF
#' @param X Data matrix (p x n)
#' @param B Number of chains
#' @param M Depth M
#' @param type "RS-PC" or "HRS-PC"
#' @param cal "pooled" or "pathwise"
#' @param n_cores Number of cores
#' @param save_raw Whether to return the raw score paths
#' @export
run_predresamp_vmf <- function(X, B = 5000, M = 50, 
                            type = c("HRS-PC", "RS-PC"), 
                            cal = c("pooled", "pathwise"), 
                            n_cores = 1, 
                            save_raw = FALSE) {
  type <- match.arg(type)
  cal  <- match.arg(cal)
  n <- ncol(X)
  p <- nrow(X)
  
  # --- Step 0: Global Setup ---
  c0 <- mle_vmf(X)
  weights <- compute_weights(n, M) # Should return w_nM and r_nM
  w_nM <- weights$W_nm
  r_nM <- weights$R_nm
  
  # --- Step 1: Simulate Raw Score Paths (Parallel) ---
  raw_results <- parallel::mclapply(1:B, function(b) {
    simulate_raw_path(c0, n, M, save_raw = save_raw)
  }, mc.cores = n_cores)
  
  # --- Step 2: Construct Calibration Estimator ---
  if (cal == "pooled") {
    Q_list <- lapply(raw_results, `[[`, "Q_nM")
    bar_Q <- Reduce("+", Q_list) / B
    I_pool <- bar_Q / w_nM
  }
  
  # --- Step 3: Return Posterior Draws ---
  # Define Post-correction helper C_epsilon
  correct_fn <- function(diff_vec, I_mat) {
    c0 + safe_sym_inverse(I_mat) %*% diff_vec
  }
  
  posterior_samples <- sapply(1:B, function(b) {
    # Get I estimator for this chain
    I_b <- if (cal == "pathwise") raw_results[[b]]$Q_nM / w_nM else I_pool
    
    D_b <- raw_results[[b]]$D_raw
    
    if (type == "RS-PC") {
      # Standard correction: c_out = C(c_nM)
      c_out <- correct_fn(D_b, I_b)
    } else {
      # Hybrid correction: c_out = C(c_nM + Z)
      # Draw Gaussian Tail Z ~ N(0, r_nM * I_b)
      tail_cov <- r_nM * I_b
      # Matrix sqrt for sampling
      e <- eigen(tail_cov)
      Z_b <- (e$vectors %*% diag(sqrt(pmax(e$values, 0)), p) %*% t(e$vectors)) %*% rnorm(p)
      c_out <- correct_fn(D_b + Z_b, I_b)
    }
    return(c_out)
  })
  
  # Transform to Polar for consistency with your previous code
  kappa_sam <- apply(posterior_samples, 2, function(v) sqrt(sum(v^2)))
  if (p == 2) {
    mu_sam <- apply(posterior_samples, 2, function(v) {
      res <- atan2(v[2], v[1])
      if (res < 0) res <- res + 2 * pi
      return(res)
    })
  } else {
    mu_sam <- sweep(posterior_samples, 2, kappa_sam, "/")
  }
  
  out <- list(
    samples = list(
      c_sam = posterior_samples,
      mu_sam = mu_sam,
      kappa_sam = kappa_sam
    ),
    setup = list(type = type, cal = cal, I_pool = if(cal=="pooled") I_pool else NULL)
  )
  
  # If requested, save the raw paths from Step 1
  if (save_raw) {
    out$raw_chains <- lapply(raw_results, `[[`, "c_chain")
  }
  
  return(out)
}
