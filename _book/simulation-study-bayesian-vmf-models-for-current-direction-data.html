6.3 Predictive Posterior via Martingale.

Score based martingale sampling without pre-conditioner (inverse info).

For efficiency, hybrid trick is adopted.

sim_martingale <- function(n, n_sim = 100,
                                         mu_true = 2, kappa_true = 4,
                                         B = 500, M = 50,
                                         seed = 1234) {
  
  A <- function(k) ifelse(k < 1e-6, 0,
                          besselI(k, 1, expon.scaled = TRUE) /
                            besselI(k, 0, expon.scaled = TRUE))
  
  score_fn <- function(eta, x) {
    k <- sqrt(sum(eta^2))
    if (k < 1e-6) return(x)
    expected <- A(k) * eta / k
    x - expected
  }
  
  rvMF2 <- function(eta) {
    k <- sqrt(sum(eta^2))
    if (k < 1e-6) {
      angle <- runif(1, 0, 2*pi)
    } else {
      mu_angle <- atan2(eta[2], eta[1])
      angle <- as.numeric(rvonmises(1, circular(mu_angle), k))
    }
    c(cos(angle), sin(angle))
  }
  
  Z_list <- list()
  Tn_vec <- c()
  times  <- numeric(n_sim)
  
  set.seed(seed)
  
  for (i in 1:n_sim) {
    theta_data <- as.numeric(rvonmises(n, circular(mu_true), kappa_true))
    
    C <- sum(cos(theta_data)); S <- sum(sin(theta_data))
    R <- sqrt(C^2 + S^2); Rbar <- R/n
    mu_hat <- atan2(S, C)
    if (Rbar < 0.001) kappa_hat <- 0
    else if (Rbar > 0.999) kappa_hat <- 1/(2*(1-Rbar))
    else kappa_hat <- uniroot(function(k) A(k) - Rbar, c(0.001, 100))$root
    eta0 <- kappa_hat * c(cos(mu_hat), sin(mu_hat))
    
    Z_mat <- matrix(0, B, 2)
    T_vec <- numeric(B)
    
    t0 <- proc.time()[3]
    
    #
    for (b in 1:B) {
      eta <- eta0
      V_mat <- matrix(0, 2, 2)   
      
      ## General iteration
      for (t in 1:M) {
        x_new <- rvMF2(eta)
        s <- score_fn(eta, x_new)        
        wt <- 1/(t + n)
        V_mat <- V_mat + wt^2 * (s %*% t(s))
        eta <- eta + wt * s              
      }
      
      ## hybird step
      w_cur <- sum(1 / ((1:M) + n)^2)
      w_rem <- sum(1 / ((M+1):(M+10000) + n)^2)
      Sigma_hat <- V_mat * (1 + w_rem / w_cur)   
      
      eig <- eigen(Sigma_hat, symmetric = TRUE)
      eig$values[eig$values < 0] <- 0
      inv_sqrt_Sigma <- eig$vectors %*% diag(1/sqrt(pmax(eig$values, 1e-12))) %*% t(eig$vectors)
      
      eps <- rnorm(2)
      eta_final <- eta + eig$vectors %*% diag(sqrt(eig$values * w_rem / (w_cur + w_rem))) %*% eps
      
      ## Calculate each stat for UQ.
      delta <- eta_final - eta0
      Z_vec <- inv_sqrt_Sigma %*% delta
      Z_mat[b, ] <- Z_vec
      T_vec[b] <- sum(Z_vec^2)  # For rough test, no rigirous proof.
    }
    t1 <- proc.time()[3]
    times[i] <- t1 - t0
    
    Z_list[[i]] <- Z_mat
    Tn_vec <- c(Tn_vec, T_vec)
  }
  
  Z_all <- do.call(rbind, Z_list)
  colnames(Z_all) <- c("Z1", "Z2")
  
  cover_Z1 <- mean(abs(Z_all[,1]) <= qnorm(0.975))
  cover_Z2 <- mean(abs(Z_all[,2]) <= qnorm(0.975))
  cover_Tn <- mean(Tn_vec <= qchisq(0.95, df = 2))
  
  list(
    Z = Z_all,
    Tn = Tn_vec,
    summary = data.frame(
      Coverage_Z1 = cover_Z1,
      Coverage_Z2 = cover_Z2,
      Coverage_Tn = cover_Tn,
      Mean_Time  = mean(times),
      stringsAsFactors = FALSE
    )
  )
}
